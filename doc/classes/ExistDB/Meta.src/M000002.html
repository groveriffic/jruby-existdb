    def named_or_ordered_options(options, ordered_options)
      hash = Hash.new
      options.each_with_index do |option, i|
        if option.is_a?(Hash) then
          hash.merge!(option)
        else
          key = ordered_options[i]
          hash[key] = option
        end
      end
      return hash
    end
