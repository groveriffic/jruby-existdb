        def delegate_to_java(*opts)
            opts.each do |opt|
                if opt.is_a?(Hash) then
                    opt.each do |to, from|
                        module_eval "
                            def #{to}(*opts)
                                ClassWrap[ @obj.#{from}( *ClassUnwrap[ opts ] ) ]
                            end
                        "
                    end
                else
                    module_eval "
                        def #{opt}(*opts)
                            ClassWrap[ @obj.#{opt}( *ClassUnwrap[ opts ] ) ]
                        end
                    "
                end
            end
        end
