    def start
      @impl = org.exist.xmldb.DatabaseImpl.new
      @properties.each{ |key, value| @impl.setProperty(key.to_s, value.to_s) }
      org.xmldb.api.DatabaseManager.registerDatabase(@impl)

      at_exit do
        stop
      end

      true
    end
